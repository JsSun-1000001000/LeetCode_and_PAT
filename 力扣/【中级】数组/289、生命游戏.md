class Solution {

public:

    void gameOfLife(vector<vector<int>>& board) {

        vector<vector<int>> copy = board;

        int x = board.size();

        int y = board[0].size();

        for(int i = 0;i < x;i++){

            for(int j = 0;j < y;j++){

                int count=0;

                if(j>0&&copy[i][j-1]) count+=1; //上

                if(j<y-1&&copy[i][j+1]) count+=1;//下

                if(i>0&&copy[i-1][j]) count+=1;//左

                if(i<x-1&&copy[i+1][j]) count+=1;//右

                if(i>0&&j>0&&copy[i-1][j-1]) count+=1;//左上

                if(i>0&&j<y-1&&copy[i-1][j+1]) count+=1; //左下

                if(i<x-1&&j>0&&copy[i+1][j-1]) count+=1;//右上

                if(i<x-1&&j<y-1&&copy[i+1][j+1]) count+=1;//右下

  

                if(count<2) board[i][j]=0;//更新细胞状态

                else if(count==3) board[i][j]=1;

                else if(count>3) board[i][j]=0;

  

            }

        }

    }

};
