## 分析

`cur = nums[i]*cur+nums[i];`

碰到二进制的部分可以考虑一下0的性质。。。
## 代码

```cpp
class Solution {

public:

    int findMaxConsecutiveOnes(vector<int>& nums) {

        int max = 0;

        int cur = 0;

        for(int i = 0; i < nums.size(); i++){

            cur = nums[i]*cur+nums[i];

            if(cur>max){

                max = cur;

            }

        }

        return max;

    }

};
```
