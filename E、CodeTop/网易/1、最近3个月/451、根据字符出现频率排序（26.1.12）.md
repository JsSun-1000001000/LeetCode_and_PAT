## 代码

按照出现频率排序：
```cpp
class Solution {

public:

    string frequencySort(string s) {

        unordered_map<char, int> mmp;

        for(auto &chr : s){

            mmp[chr]++;

        }

        vector<pair<char, int>>vec;

        for(auto &it : mmp){

            vec.emplace_back(it);

        }

        sort(vec.begin(),

            vec.end(),

            [](const pair<char, int> &a,

                const pair<char, int> &b){

                    return a.second > b.second;

                });

        string ret;

        for(auto &[chr, num] : vec){

            for(int i = 0; i < num; i++){

                ret.push_back(chr);

            }

        }

        return ret;

    }

};
```
