## 分析 Analysis Section

复习老师讲过的回溯模板：
```cpp
void backtracking(数据, 结果){
	if(结束条件){
		结果收集
		return;
	}
	for(可能性选择){
		选择
		backtracking();
		撤销选择
	}
}
```
## 代码 Code

```cpp
class Solution {
public:
    //全排列 nums 结果集数组 哪个位置后开始进行全排列处理
    void backtracking(vector<int>&nums, vector<vector<int> > &ans, int begin){
        //检查 结束了吗 个数与长度相等了吗
        if(begin == nums.size()){
            ans.push_back(nums);
            return;
        }
        //遍历剩余可能性
        for(int i = begin; i < nums.size(); i++){
            //选择
            swap(nums[i], nums[begin]);
            //对剩余元素全排列
            backtracking(nums, ans, begin+1);
            //撤销选择
            swap(nums[i], nums[begin]);
        }
    }
    vector<vector<int>> permute(vector<int>& nums) {  
        //结果集数组 二维数组
        vector<vector<int> > ans;
        //哪个位置后开始进行全排列处理
        backtracking(nums, ans,  0);
        return ans;
    }
};
```
