## 分析

用reverse函数反转数组；
## 代码
```cpp
class Solution {

public:

    void rotate(vector<int>& nums, int k) {

        k = k % nums.size();//防止k比n大

        reverse(nums.begin(), nums.end());

        reverse(nums.begin(), nums.begin()+k);

        reverse(nums.begin()+k, nums.end());

    }

};
```

class Solution {

public:

    void rotate(vector<int>& nums, int k) {

        //自己实现reverse

        auto Reverse = [&](int i, int j){

            while(i<j){

                swap(nums[i++],nums[j--]);

            }

        };

        int n =nums.size();

        k = k % n;

        Reverse(0, n-1);

        Reverse(0, k-1);

        Reverse(k, n-1);

    }

};
