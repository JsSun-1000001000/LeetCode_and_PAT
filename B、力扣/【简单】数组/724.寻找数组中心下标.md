class Solution {

public:

    int pivotIndex(vector<int>& nums) {

        //数组的总和

        int total = accumulate(nums.begin(), nums.end(), 0);

        //index左侧数组的总和

        int leftsum = 0;

        //遍历数组

        for( int i = 0; i < nums.size(); ++i){

            //如果leftsum == total - leftsum - leftsum[i]说明是最左侧的结果

            if(2 * leftsum + nums[i] == total){

                return i;

            }

            //不是就把当前元素加入sum中

            leftsum = leftsum + nums[i];

        }

        return -1;

    }

};
