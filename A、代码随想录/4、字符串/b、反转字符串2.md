## [541. 反转字符串 II - 力扣（LeetCode）](https://leetcode.cn/problems/reverse-string-ii/description/)

## 代码

```cpp
class Solution {

public:

    string reverseStr(string s, int k) {

        for(int i = 0; i < s.size(); i+=(2*k)){

            if(i + k <= s.size()){

                reverse(s.begin() + i, s.begin() + i + k);

            }

            else{

                reverse(s.begin() + i, s.end());

            }

        }

        return s;

    }

};
```
